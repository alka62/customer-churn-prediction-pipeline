        "import numpy as np\n",

        "# ============================================================\n",
        "df = df.dropna()\n",

        "# ============================================================\n",
        "# 3. IDENTIFY FEATURES\n",
        "# ============================================================\n",
        "target_col = \"Churn\"\n",
        "\n",
        "categorical_cols = df.select_dtypes(include=[\"object\"]).columns.tolist()\n",
        "if target_col in categorical_cols:\n",
        "    categorical_cols.remove(target_col)\n",
        "\n",
        "numerical_cols = df.select_dtypes(include=np.number).columns.tolist()\n",
        "\n",
        "# ============================================================\n",
        "# 4. ENCODING METHODS\n",
        "# ============================================================\n",
        "\n",
        "# Label Encoding on first categorical column\n",
        "if len(categorical_cols) > 0:\n",
        "    le = LabelEncoder()\n",
        "    df[categorical_cols[0]] = le.fit_transform(df[categorical_cols[0]])\n",
        "\n",
        "    # One-Hot Encoding on remaining categorical columns\n",
        "    df = pd.get_dummies(df, columns=categorical_cols[1:], drop_first=True)\n",
        "\n",
        "    # Ordinal Encoding example (if any categorical columns left)\n",
        "    remaining_cat = df.select_dtypes(include=[\"object\"]).columns.tolist()\n",
        "    if target_col in remaining_cat:\n",
        "        remaining_cat.remove(target_col)\n",
        "\n",
        "    for col in remaining_cat:\n",
        "        oe = OrdinalEncoder()\n",
        "        df[[col]] = oe.fit_transform(df[[col]])\n",
        "\n",
        "# ============================================================\n",
        "# 5. FEATURE ENGINEERING (Create 5 new features)\n",
        "# ============================================================\n",
        "num_cols = df.select_dtypes(include=np.number).columns.tolist()\n",
        "if target_col in num_cols:\n",
        "    num_cols.remove(target_col)\n",
        "\n",
        "if len(num_cols) >= 2:\n",
        "    df[\"feature_sum\"] = df[num_cols[0]] + df[num_cols[1]]\n",
        "    df[\"feature_diff\"] = df[num_cols[0]] - df[num_cols[1]]\n",
        "    df[\"feature_product\"] = df[num_cols[0]] * df[num_cols[1]]\n",
        "    df[\"feature_ratio\"] = df[num_cols[0]] / (df[num_cols[1]] + 1)\n",
        "    df[\"feature_square\"] = df[num_cols[0]] ** 2\n",

        "    df[col] = np.where(df[col] < lower, lower, df[col])\n",
        "    df[col] = np.where(df[col] > upper, upper, df[col])\n",

        "# ============================================================\n",
        "# 7. PREPARE DATA\n",
        "# ============================================================\n",

        "    y = df[target_col]\n",

        "# ============================================================\n",
        "# 8. SCALING\n",

        "# 9. TRAIN TEST SPLIT\n",

        ")\n",
        "\n",
        "# ============================================================\n",
        "# 10. BUILD PIPELINE\n",
        "# ============================================================\n",
        "pipeline = Pipeline([\n",

        "])\n",
